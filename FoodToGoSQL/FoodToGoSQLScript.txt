***DATABASE
Create Database FoodToGoDB;

Use FoodToGoDB;


**CONSUMER

create table Consumer (
   Consumer_ID INT primary key auto_increment not NULL,
   Consumer_Fname varchar(50) not NULL,
   Consumer_Lname varchar(50) not NULL,  
   Consumer_Email varchar(50) not NULL, 
   Consumer_password varchar(15) not NULL,  
   Consumer_Number varchar(15) not NULL,
   Consumer_Billing_Address varchar(200) not NULL
);


**RESTAURANT

create table Restaurant (
   Restaurant_ID INT primary key auto_increment NOT NULL,
   Restaurant_Email varchar(50) NOT NULL,
   Restaurant_password varchar(15) NOT NULL,
   Restaurant_BranchName varchar(50) NOT NULL, 
   Restaurant_Number varchar(15) NOT NULL,
   Restaurant_BranchAddress varchar(200) NOT NULL
);


** Courier

create table Courier (
   Courier_ID INT primary key auto_increment NOT NULL,
   Courier_Fname varchar(50) NOT NULL,
   Courier_Lname varchar(50) NOT NULL,  
   Courier_Email varchar(50) NOT NULL,
   Courier_password varchar(15) NOT NULL,  
   Courier_Number varchar(15) NOT NULL,
   Courier_Vehicle varchar(15) NOT NULL
);


** Inventory

CREATE TABLE Inventory (
   Inventory_ID int NOT NULL AUTO_INCREMENT PRIMARY KEY auto_increment,
   Restaurant_ID INT,
   FOREIGN KEY(Restaurant_ID)  REFERENCES Restaurant(Restaurant_ID),
   Food_Category varchar(15) NOT NULL,
   Food_Item varchar(15) NOT NULL,
   Quantity int (7)
);


** Customer_Order

CREATE TABLE Customer_Order(
    Order_ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Restaurant_ID int,
    FOREIGN KEY(Restaurant_ID)  REFERENCES Restaurant(Restaurant_ID),
    Food_Item varchar (15), 
    Order_Price decimal(7,2),
    Order_Status varchar(20) NOT NULL
);


**DELIVERY

CREATE TABLE Delivery (
    Delivery_ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Courier_ID int,
    FOREIGN KEY (Courier_ID)  REFERENCES Courier(Courier_ID)
    Total_Order_Price decimal(9,2),
    Delivery_Fee decimal(5,2),
    Delivery_Status varchar(20)
);



**BILLING

CREATE TABLE Billing (
    Billing_ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Consumer_ID int,
    FOREIGN KEY (Consumer_ID) REFERENCES Consumer(Consumer_ID)
);



**PAYMENT

CREATE TABLE Payment (
    Payment_ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Delivery_ID int,
    FOREIGN KEY (Delivery_ID) REFERENCES Delivery(Delivery_ID),
    Payment_Type varchar(20),
    Payment_Status varchar(20)
);


**NOTIFICATION

CREATE TABLE Notification (
    Notif_ID int NOT NULL AUTO_INCREMENT PRIMARY KEY,
    Delivery_ID int,
    FOREIGN KEY (Delivery_ID) REFERENCES Delivery(Delivery_ID),
    Order_ID int,
    FOREIGN KEY (Order_ID) REFERENCES Customer_Order(Order_ID),
    Restaurant_ID int,
    FOREIGN KEY (Restaurant_ID) REFERENCES Restaurant(Restaurant_ID),
    Consumer_ID int,
    FOREIGN KEY (Consumer_ID) REFERENCES Consumer(Consumer_ID),
    Courier_ID int,
    FOREIGN KEY (Courier_ID) REFERENCES Courier(Courier_ID),
    Order_Status varchar(20),
    Delivery_Status varchar(20)
 );
